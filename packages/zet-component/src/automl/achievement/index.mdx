## Basic usage

<Playground style={{backgroundColor:'aliceblue'}}>
  <State initial={{ show:'chart' }}>
    {({ state, setState }) => {
      const { Achievement } = AutoML;
      const { Panel } = Achievement;
      const changeHandle = (option, record)=>{
         console.log('record',option,record);
         if(option=='rotate'){
            setState({
              show:record.type
            })
         }
      }
      const {show} = state;
      return <Achievement title='Test' extraKeys={['1']} style={{height:500}} onChange={changeHandle}>
        <Panel option='1' height={300}>{show=='chart' ? <Example1 /> : <Example2/>}</Panel>
        <Panel option='2' flex >{show=='chart' ? <Example1 style={{height:600}}/> : <Example2 style={{height:600}}/>}</Panel>
      </Achievement>
    }}
  </State>
</Playground>
